# Give non-feudal Barony to female character; abdicate to spouse if liege is not player

character_event = {
	id = amareturn.801
	desc = amareturn.801
	
	only_rulers = yes
	only_men = yes
	
	trigger = {
		ai = yes
		OR = {
			AND = {
				is_feudal = no
				religion_group = amazonian
				num_of_baron_titles = 1
				lower_tier_than = COUNT
			}
			AND = {
				religion_group = amazonian
				liege = {
					ai = yes
				}
				is_married = yes
				primary_spouse = {
					religion_group = amazonian
				}
			}
		}
		is_female = no
		any_courtier = { is_female = yes }
	}
		
	mean_time_to_happen = {
		days = 1
	}
		
	option = {
		name = OK			
		# Abdicate to spouse, if married
		if = {
			limit = {
				is_married = yes
				any_spouse = {
					is_female = yes
					religion_group = amazonian
				}
			}
			random_spouse = {
				limit = {
					is_female = yes
					religion_group = amazonian
				}
				ROOT = {
					abdicate_to = PREV
				}
			}
		}
		
		# Otherwise, to female courtier
		if = {
			limit = {
				OR = {
					is_married = no
					NOT = {
						any_spouse = {
							is_female = yes
							religion_group = amazonian
						}
					}
				}
			}
			random_courtier = {
				limit = {
					is_female = yes
					religion_group = amazonian
				}
				ROOT = {
					abdicate = PREV
				}
			}
			}
		}
	}

# Abdicate to spouse

character_event = {
	id = amareturn.900
	desc = amareturn.900
	
	only_men = yes
	only_rulers = yes
	
	trigger = {
		ai = yes
		liege = { #Player can grant to men if he really wants to
			ai = yes
			religion_group = amazonian
		}
		primary_title = {
			is_recent_grant = yes
		}
		is_female = no
		is_married = yes
		spouse = { #Player won't receive free counties and titles this way
			ai = yes
			religion_group = amazonian
		}
		religion_group = amazonian
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = OK
		abdicate = yes
		abdicate_to = spouse
		move_character = spouse
	}
}
